Cross match.¶

Find the nearest source.

If it is a star within a radius of 2 times the PSF, in r-band, that's the match.

If it is a galaxy within a radius of 5 times the half-light radius, then that is the match.

Flag instances where a star also had a galaxy match, and save the ls_id.

# 0 if nothing within 72 arcsec; otherwise the number of nearby things
cand_region_pop = np.zeros(len(cand_id), dtype='int')

# -1 if no match; otherwise, the index in the dr10 array
cand_match_idx = np.zeros(len(cand_id), dtype='int') - 1

# -1 if no info; 0 if galaxy; 1 if star
cand_match_star = np.zeros(len(cand_id), dtype='int') - 1

# 1 if galaxy and potentially nuclear; 0 otherwise
cand_match_gnuc = np.zeros(len(cand_id), dtype='int')

# -1 if not matched; 0 if only matched to either star or galaxy; 1 if one or more of each
cand_match_flag = np.zeros(len(cand_id), dtype='int') - 1

# value of the offset in arcsec
cand_match_offset = np.zeros(len(cand_id), dtype='float')

for c, coord in enumerate(cand_coords):
    verbose = False
    tx = np.where(cand_id[c] == templist)[0]
    if len(tx) == 1:
        verbose = True
    
    ralo = cand_ra[c] - 0.02
    rahi = cand_ra[c] + 0.02
    delo = cand_dec[c] - 0.02
    dehi = cand_dec[c] + 0.02
    tx = np.where((dr10_ra > ralo) & (dr10_ra < rahi) & 
                  (dr10_dec > delo) & (dr10_dec < dehi))[0]
    del ralo, rahi, delo, dehi
    
    cand_region_pop[c] = len(tx)
    
    if len(tx) > 0:
    
        off_sky = coord.separation(dr10_coords[tx]).arcsec
        off_psf = off_sky / dr10_psfsize_r[tx]
        off_hlr = off_sky / dr10_shape_r[tx]
        
        ### identify all the potential star or galaxy matches
        sx = np.where((off_psf < 2) & (dr10_type[tx] == 'PSF'))[0]
        gx = np.where((off_hlr < 5) & (dr10_type[tx] != 'PSF'))[0]
        
        ### if there are some of each, flag as potential multiple matches
        if (len(sx) > 0) | (len(gx) > 0):
            cand_match_flag[c] = 0
        if (len(sx) > 0) & (len(gx) > 0):
            cand_match_flag[c] = 1
        
        ### print for a few examples
        if (len(sx) > 0) | (len(gx) > 0):
            if verbose:
                print(' ')
                print(cand_id[c], cand_ra[c], cand_dec[c])
                print(cand_LSlink[c])
                print(cand_DVlink[c])
                print(len(sx), len(gx))
                if len(sx) > 0:
                    for x in sx:
                        print('star ', x, tx[x], dr10_type[tx[x]], off_sky[x], off_psf[x], off_hlr[x], 
                              dr10_psfsize_r[tx[x]], dr10_shape_r[tx[x]], dr10_mag_r[tx[x]])
                if len(gx) > 0:
                    for x in gx:
                        print('gal ', x, tx[x], dr10_type[tx[x]], off_sky[x], off_psf[x], off_hlr[x], 
                              dr10_psfsize_r[tx[x]], dr10_shape_r[tx[x]], dr10_mag_r[tx[x]])
        del sx, gx
        
        ### sort by separation
        ix = np.argsort(off_sky)
        ### if the most nearby object is a star
        if dr10_type[tx[ix[0]]] == 'PSF':
            ### and that star is within 2x PSF, it's a match
            if off_psf[ix[0]] <= 2:
                cand_match_idx[c] = tx[ix[0]]
                cand_match_star[c] = 1
                cand_match_offset[c] = off_sky[ix[0]]
        del ix
        
        ### if not a star, check for a galaxy
        if cand_match_star[c] < 1:
            gx = np.where((off_hlr < 5) & (dr10_type[tx] != 'PSF'))[0]
            if len(gx) > 0:
                ### take the most nearby galaxy
                mx = np.argmin(off_sky[gx])
                cand_match_idx[c] = tx[gx[mx]]
                cand_match_star[c] = 0
                cand_match_offset[c] = off_sky[gx[mx]]
                if off_sky[gx[mx]] < dr10_psfsize_r[tx[gx[mx]]]:
                    cand_match_gnuc[c] = 1
                del mx
            del gx
            
        if verbose:
            print(cand_region_pop[c], cand_match_flag[c], cand_match_idx[c], 
                  cand_match_star[c], cand_match_gnuc[c])
    del tx

for c in range(len(cand_id)):
    if cand_match_star[c] == 0:
        host_ra = dr10_ra[cand_match_idx[c]]
        host_dec = dr10_dec[cand_match_idx[c]]
        tx = np.where((np.abs(dr9_ra - host_ra) < 0.02) & 
                      (np.abs(dr9_dec - host_dec) < 0.02))[0]
        if len(tx) > 0:
            coord_host = SkyCoord(host_ra, host_dec, unit="deg")
            coords_dr9 = SkyCoord(dr9_ra[tx], dr9_dec[tx], unit="deg")
            offsets = coord_host.separation(coords_dr9).arcsec
            ox = np.argmin(offsets)
            if offsets[ox] <= 2.0:
                cand_dr9_zphot_mean[c] = dr9_zphot_mean[tx[ox]]
                cand_dr9_zphot_std[c] = dr9_zphot_std[tx[ox]]
                cand_dr9_zspec[c] = dr9_zspec[tx[ox]]
            del ox, offsets, coords_dr9, coord_host
        del host_ra, host_dec, tx
